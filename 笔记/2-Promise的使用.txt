1. promise 是一个对象 对象中有一个方法 .then
2. 通过then方法,就可以得到成功的或失败的结果
3. then需要传递两个回调函数，如果Promise成功了，调用第一个回调函数
4. 如果promise失败了，调用第二个回调函数
5. 掉第一个，这个回调函数的参数，就是成功的结果
6. 掉第二个，这个回调函数的参数，就是失败的结果

<script>
  function execCode(counter) {
    let promise = new Promise((resolve, reject) => {
      // 在执行器里写异步代码
      setTimeout(() => {
        if (counter > 0) {
          let t = 0;
          for (let i = 0; i <= counter; i++) {
            t += i;
          }
          resolve(t);
        } else {
          reject(`${counter}有问题`);
        }
      }, 300);
    });
    return promise;
  }
  // promise 是一个对象 对象中有一个方法 .then
  // 通过then方法,就可以得到成功的或失败的结果
  // then需要传递两个回调函数，如果Promise成功了，调用第一个回调函数
  // 如果promise失败了，调用第二个回调函数
  // 掉第一个，这个回调函数的参数，就是成功的结果
  // 掉第二个，这个回调函数的参数，就是失败的结果
  let promise = execCode(100);
  promise.then((resp)=>{
      console.log(resp);
  },(err)=>{
    console.log(err);
  })
</script>
